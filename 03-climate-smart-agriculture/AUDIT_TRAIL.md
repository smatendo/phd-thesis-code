    "from pathlib import Path\n",
    "import warnings\n",
    "\n",
    "PROJECT_DIR = Path.cwd()\n",
    "if PROJECT_DIR.name in {\"notebooks\", \"needs_verification\"}:\n",
    "    PROJECT_DIR = PROJECT_DIR.parent\n",
    "\n",
    "DATA_FILE = PROJECT_DIR / \"data\" / \"questionnaire.xlsx\"\n",
    "OUTPUTS_DIR = PROJECT_DIR / \"outputs\"\n",
    "\n",
    "if not DATA_FILE.exists():\n",
    "    raise FileNotFoundError(\n",
    "        f\"Input file not found: {DATA_FILE}\\n\"\n",
    "        \"Place the anonymised questionnaire file in data/questionnaire.xlsx \"\n",
    "        \"or update DATA_FILE.\"\n",
    "    )\n",
    "\n",
    "OUTPUTS_DIR.mkdir(parents=True, exist_ok=True)\n",
    "\n",
    "import pandas as pd\n",
    "import matplotlib.pyplot as plt\n",
    "\n",
    "OUT_DIR = OUTPUTS_DIR / \"04_farm_size_gender\"\n",
    "OUT_DIR.mkdir(parents=True, exist_ok=True)\n",
    "\n",
    "df = pd.read_excel(DATA_FILE)\n",
    "df.columns = df.columns.str.strip()\n",
    "\n",
    "gender_col = \"What is your gender?\"\n",
    "farm_size_col = \"What is the size of your farm (in acres)?\"\n",
    "\n",
    "for column in [gender_col, farm_size_col]:\n",
    "    if column not in df.columns:\n",
    "        raise KeyError(f\"Missing questionnaire column: {column}\")\n",
    "\n",
    "df[\"female\"] = (\n",
    "    df[gender_col].astype(\"string\").str.strip().str.lower().eq(\"female\").astype(int)\n",
    ")\n",
    "\n",
    "df[\"farm_size_acres\"] = (\n",
    "    df[farm_size_col].astype(\"string\")\n",
    "      .str.replace(\",\", \"\", regex=False)\n",
    "      .str.extract(r\"(\\d+(?:\\.\\d+)?)\", expand=False)\n",
    "      .pipe(pd.to_numeric, errors=\"coerce\")\n",
    ")\n",
    "df[\"farm_size_ha\"] = df[\"farm_size_acres\"] * 0.40468564224\n",
    "\n",
    "plot_data = df.dropna(subset=[\"female\", \"farm_size_ha\"]).copy()\n",
    "men = plot_data.loc[plot_data[\"female\"] == 0, \"farm_size_ha\"]\n",
    "women = plot_data.loc[plot_data[\"female\"] == 1, \"farm_size_ha\"]\n",
    "\n",
    "fig, ax = plt.subplots(figsize=(6, 4))\n",
    "ax.violinplot(\n",
    "    [men, women],\n",
    "    showmeans=True,\n",
    "    showmedians=False,\n",
    "    showextrema=True,\n",
    ")\n",
    "ax.set_xticks([1, 2])\n",
    "ax.set_xticklabels([\"Men\", \"Women\"])\n",
    "ax.set_ylabel(\"Farm size (ha)\")\n",
    "ax.grid(axis=\"y\", linestyle=\"--\", alpha=0.4)\n",
    "plt.tight_layout()\n",
    "\n",
    "fig.savefig(OUT_DIR / \"figure_4_26_farm_size_by_gender.png\", dpi=300, bbox_inches=\"tight\")\n",
    "fig.savefig(OUT_DIR / \"figure_4_26_farm_size_by_gender.pdf\", bbox_inches=\"tight\")\n",
    "plt.show()\n"
