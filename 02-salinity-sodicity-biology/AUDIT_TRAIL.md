    "# Candidate models reported in Table 4.5\n",
    "formulas = {\n",
    "    \"M1_EC\": \"log_Kfs ~ EC12 + C(Zone)\",\n",
    "    \"M2_ESP\": \"log_Kfs ~ ESP + C(Zone)\",\n",
    "    \"M3_SOC_pH\": \"log_Kfs ~ SOC + pH + C(Zone)\",\n",
    "    \"M4_full\": \"log_Kfs ~ EC12 + ESP + SOC + Clay + pH + C(Zone)\",\n",
    "}\n",
    "\n",
    "models = {}\n",
    "metrics_rows = []\n",
    "coefficient_rows = []\n",
    "\n",
    "for model_name, formula in formulas.items():\n",
    "    model = smf.ols(formula=formula, data=model_data).fit()\n",
    "    models[model_name] = model\n",
    "\n",
    "    metrics_rows.append({\n",
    "        \"Model\": model_name,\n",
    "        \"Formula\": formula,\n",
    "        \"N\": int(model.nobs),\n",
    "        \"R2\": model.rsquared,\n",
    "        \"Adjusted_R2\": model.rsquared_adj,\n",
    "        \"AIC\": model.aic,\n",
    "        \"BIC\": model.bic,\n",
    "        \"p_value\": model.f_pvalue,\n",
    "    })\n",
    "\n",
    "    confidence_intervals = model.conf_int()\n",
    "    for term in model.params.index:\n",
    "        coefficient_rows.append({\n",
    "            \"Model\": model_name,\n",
    "            \"Variable\": term,\n",
    "            \"Coefficient\": model.params[term],\n",
    "            \"SE\": model.bse[term],\n",
    "            \"p_value\": model.pvalues[term],\n",
    "            \"CI_low\": confidence_intervals.loc[term, 0],\n",
    "            \"CI_high\": confidence_intervals.loc[term, 1],\n",
    "        })\n",
    "\n",

    "ols_model = models[\"M3_SOC_pH\"]\n",

    "        })\n",
